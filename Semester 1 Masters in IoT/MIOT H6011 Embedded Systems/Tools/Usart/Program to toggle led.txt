#include <avr/io.h>
#include <util/delay.h>

#include "usart.h"


int main(void)
{
	usartInit();

	//setup code goes here
	DDRB |= (1<<5);

	while(1)
	{
		if(usartCharReceived())
		{
			char rxChar = usartReadChar();

			if(rxChar == 'a')
			{
				PORTB |= (1<<5);
			}
			else if(rxChar == 'b')
			{
				PORTB &= ~(1<<5);
			}
		}
	}

}
